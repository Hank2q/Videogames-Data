CREATE TABLE "game_genre"("genre_id" integer NOT NULL, "game_id" integer NOT NULL,CONSTRAINT "FK_40" FOREIGN KEY("genre_id") REFERENCES "genres" ("id"), CONSTRAINT "FK_43" FOREIGN KEY("game_id") REFERENCES "games" ("id"))
CREATE INDEX "fkIdx_40" ON "game_genre"("genre_id")
CREATE INDEX "fkIdx_43" ON "game_genre"("game_id")
CREATE TABLE "games"("id" integer NOT NULL, "title" varchar(1024) NOT NULL, "meta_score" integer NULL, "user_score" real NULL, "developer_id" integer NULL, "rating_id" integer NULL,"release_date" date NULL, "platform_id" integer NOT NULL, CONSTRAINT "PK_games" PRIMARY KEY("id"), CONSTRAINT "FK_19" FOREIGN KEY("developer_id") REFERENCES "developers" ("id"),CONSTRAINT "FK_26" FOREIGN KEY("rating_id") REFERENCES "ratings" ("id"), CONSTRAINT "FK_50"  FOREIGN KEY("platform_id") REFERENCES "platforms" ("id"))
CREATE INDEX "fkIdx_19" ON "games"("developer_id")
CREATE INDEX "fkIdx_26" ON "games"("rating_id")
CREATE INDEX "fkIdx_50" ON "games"("platform_id")
CREATE TABLE "genres"("id" integer NOT NULL, "name" varchar(255) NOT NULL,CONSTRAINT "PK_genres" PRIMARY KEY("id"))
CREATE TABLE "platforms"("id" integer NOT NULL, "name" varchar(50) NOT NULL, CONSTRAINT "PK_platforms" PRIMARY KEY("id"))
CREATE TABLE "ratings"("id" integer NOT NULL, "name" varchar(10) NOT NULL, CONSTRAINT "PK_ratings" PRIMARY KEY("id"))
CREATE TABLE "developers"("id" integer NOT NULL, "name" varchar(255) NOT NULL, CONSTRAINT "PK_developers" PRIMARY KEY("id"))
